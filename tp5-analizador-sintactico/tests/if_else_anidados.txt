program if_else_anidados;

var x, y, z: integer;

begin
  x := 5;
  y := 10;
  z := 0;

  if x > 0 then
    if y > 0 then
      if z = 0 then
        x := 1
      else
        x := 2
    else
      x := 3
  else
    x := 4;

  if x > 0 then
    if y > 5 then
      y := y + 1
    else
      y := y - 1
  else
    z := z + 1
end.
